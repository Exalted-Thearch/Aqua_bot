/*
 * Assigns 3 random quests to a user if they don't have active, uncompleted quests
 * or if their quests are from a previous day.
 * @param {string} userId The ID of the user.
 * @returns {Promise<Array>} A promise that resolves with the assigned quests.
 */
// index.js (or wherever assignDailyQuests is defined)

async function assignDailyQuests(userId) {
  const today = new Date().toISOString().slice(0, 10); // YYYY-MM-DD

  return new Promise((resolve, reject) => {
    db.all(
      `SELECT * FROM user_quests WHERE user_id = ? AND assigned_date = ? AND completed = 0`,
      [userId, today],
      async (err, activeQuests) => {
        if (err) return reject(err);

        // If user already has 3 active (uncompleted) quests for today, just return them
        if (activeQuests.length === 3) {
          return resolve(activeQuests);
        }

        // If we reach here, it means the user needs new quests or new quests have been assigned but some were completed.
        // We need to clear ALL quests for this user for TODAY that are either:
        // 1. Not yet completed (but we are re-assigning, implying they want new ones)
        // 2. Completed AND Claimed (they should be cleared to make space for new ones)
        // We should NOT delete completed but unclaimed quests, as they still need to be claimed.

        // Strategy: Delete all of today's quests that are either not completed OR are completed AND claimed.
        // This ensures only "completed but unclaimed" quests remain if they exist for today.
        db.run(
          `DELETE FROM user_quests WHERE user_id = ? AND assigned_date = ? AND (completed = 0 OR claimed = 1)`,
          [userId, today],
          async (err) => {
            if (err) {
              console.error(
                "Error clearing old quests for today:",
                err.message
              );
              return reject(err);
            }

            // Also delete any quests from previous days, completed or not.
            db.run(
              `DELETE FROM user_quests WHERE user_id = ? AND assigned_date != ?`,
              [userId, today],
              async (err) => {
                if (err) {
                  console.error(
                    "Error clearing quests from previous days:",
                    err.message
                  );
                  return reject(err);
                }

                // Now, fetch any remaining quests for today that are completed but NOT claimed.
                db.all(
                  `SELECT * FROM user_quests WHERE user_id = ? AND assigned_date = ? AND completed = 1 AND claimed = 0`,
                  [userId, today],
                  async (err, existingCompletedUnclaimedQuests) => {
                    if (err) return reject(err);

                    // We need to assign enough new quests to bring the total up to 3,
                    // considering any already existing completed but unclaimed quests.
                    const questsToAssignCount =
                      3 - existingCompletedUnclaimedQuests.length;

                    if (questsToAssignCount <= 0) {
                      // User already has 3 completed (unclaimed) quests, or more.
                      // This scenario should be rare with the current logic, but a safeguard.
                      return resolve(existingCompletedUnclaimedQuests);
                    }

                    // Filter out quest types that are already existing as completed/unclaimed for today
                    const existingQuestTypes =
                      existingCompletedUnclaimedQuests.map((q) => q.quest_type);
                    const availableQuests = quests.filter(
                      (q) => !existingQuestTypes.includes(q.type)
                    );

                    const shuffledQuests = [...availableQuests].sort(
                      () => 0.5 - Math.random()
                    );
                    const assignedNew = shuffledQuests.slice(
                      0,
                      questsToAssignCount
                    );

                    const stmt = db.prepare(
                      `INSERT INTO user_quests (user_id, quest_type, target_count, current_progress, assigned_date, completed, claimed) VALUES (?, ?, ?, ?, ?, 0, 0)`
                    );
                    for (const quest of assignedNew) {
                      stmt.run(userId, quest.type, quest.target, 0, today);
                    }
                    stmt.finalize((err) => {
                      if (err) return reject(err);

                      // Fetch and return all quests for today (newly assigned + existing completed-unclaimed)
                      db.all(
                        `SELECT * FROM user_quests WHERE user_id = ? AND assigned_date = ?`,
                        [userId, today],
                        (err, allTodayQuests) => {
                          if (err) return reject(err);
                          resolve(allTodayQuests);
                        }
                      );
                    });
                  }
                );
              }
            );
          }
        );
      }
    );
  });
}


// Merged client.on("messageCreate") blocks
client.on("messageCreate", async (message) => {
  if (message.author.bot) return; // Ignore bot messages

  const userId = message.author.id;
  const today = new Date().toISOString().slice(0, 10);

  // Function to update quest progress
  async function updateQuestProgress(questType, increment = 1) {
    return new Promise((resolve, reject) => {
      db.get(
        `SELECT * FROM user_quests WHERE user_id = ? AND quest_type = ? AND assigned_date = ? AND completed = 0`,
        [userId, questType, today],
        (err, row) => {
          if (err) return reject(err);
          if (row) {
            const newProgress = row.current_progress + increment;
            const isCompleted = newProgress >= row.target_count ? 1 : 0;

            db.run(
              `UPDATE user_quests SET current_progress = ?, completed = ? WHERE user_id = ? AND quest_type = ? AND assigned_date = ?`,
              [newProgress, isCompleted, userId, questType, today],
              function (err) {
                if (err) return reject(err);
                if (isCompleted && !row.completed) {
                  // If just completed
                  message.channel.send(
                    `🎉 Quest Completed! You finished "${
                      quests.find((q) => q.type === questType).description
                    }"! Use \`${prefix}claimquests\` to claim your rewards.`
                  );
                }
                resolve();
              }
            );
          } else {
            resolve(); // No active quest of this type
          }
        }
      );
    });
  }

  // --- Quest Tracking Logic ---

  // Track messages
  await updateQuestProgress("send_messages");

  // Track images (attachments that are images)
  if (message.attachments.size > 0) {
    const imageAttachments = message.attachments.filter(
      (attachment) =>
        attachment.contentType && attachment.contentType.startsWith("image/")
    );
    if (imageAttachments.size > 0) {
      await updateQuestProgress("send_images", imageAttachments.size);
    }
  }

  // Track emojis (custom and unicode)
  const customEmojiRegex = /<a?:\w+:\d+>/g; // Matches custom Discord emoji
  const customEmojis = message.content.match(customEmojiRegex) || [];
  const emojiCountFromRegex = (message.content.match(emojiRegex()) || [])
    .length;

  const emojiCount = customEmojis.length + emojiCountFromRegex;

  if (emojiCount > 0) {
    await updateQuestProgress("send_emojis", emojiCount);
  }

  // Track stickers
  if (message.stickers.size > 0) {
    await updateQuestProgress("send_stickers", message.stickers.size);
  }

  // Track invite links (basic check, could be improved)
  const inviteLinkRegex =
    /(discord\.gg\/|discordapp\.com\/invite\/)([a-zA-Z0-9]+)/g;
  const inviteLinks = message.content.match(inviteLinkRegex) || [];
  if (inviteLinks.length > 0) {
    await updateQuestProgress("send_invites", inviteLinks.length);
  }

  // --- Command Handling Logic ---
  const args = message.content.slice(prefix.length).trim().split(/ +/);
  const command = args.shift().toLowerCase();

  // Command: !quests
  if (command === "quests") {
    try {
      const userQuests = await assignDailyQuests(userId); // Ensure quests are assigned/fetched

      if (userQuests.length === 0) {
        return message.channel.send(
          "You currently have no active quests. Try again later!"
        );
      }

      let questMessage = `**Daily Quests for ${message.author.username}:**\n\n`;
      let allCompletedAndClaimed = true; // Track if all are completed AND claimed

      userQuests.forEach((q) => {
        const questDef = quests.find((qd) => qd.type === q.quest_type);
        if (questDef) {
          const status = q.completed ? "✅ COMPLETED" : "⏳ IN PROGRESS";
          const claimedStatus = q.claimed ? "(Claimed)" : "";
          questMessage += `- **${questDef.description}**: ${q.current_progress}/${q.target_count} ${status} ${claimedStatus}\n`;
          if (!q.completed || !q.claimed) {
            // If not completed OR not claimed, then allCompletedAndClaimed is false
            allCompletedAndClaimed = false;
          }
        }
      });

      // Check if all quests are completed AND not yet claimed
      const totalUnclaimedCompletedQuests = userQuests.filter(
        (q) => q.completed && !q.claimed
      ).length;

      if (totalUnclaimedCompletedQuests === 3) {
        questMessage += `\n🌟 All quests completed! Use \`${prefix}claimquests\` to claim your rewards and bonus!`;
      } else if (
        totalUnclaimedCompletedQuests > 0 &&
        totalUnclaimedCompletedQuests < 3
      ) {
        questMessage += `\nUse \`${prefix}claimquests\` to claim rewards for completed quests.`;
      } else if (allCompletedAndClaimed) {
        // Check the new variable
        questMessage +=
          "\nAll your daily quests are completed and claimed! Come back tomorrow for new quests.";
      } else if (
        totalUnclaimedCompletedQuests === 0 &&
        userQuests.some((q) => !q.completed)
      ) {
        questMessage += "\nKeep working on your quests!";
      }

      message.channel.send(questMessage);
    } catch (error) {
      console.error("Error fetching quests:", error.message);
      message.channel.send("An error occurred while fetching your quests.");
    }
  }

  // Command: !claimquests
  if (command === "claimquests") {
    let totalPointsEarned = 0;
    let bonusAwarded = false;

    db.all(
      `SELECT * FROM user_quests WHERE user_id = ? AND assigned_date = ? AND completed = 1 AND claimed = 0`,
      [userId, today],
      async (err, completedUnclaimedQuests) => {
        if (err) {
          console.error(err.message);
          return message.channel.send(
            "An error occurred while trying to claim rewards."
          );
        }

        if (completedUnclaimedQuests.length === 0) {
          return message.channel.send(
            "You have no completed and unclaimed quests for today."
          );
        }

        for (const q of completedUnclaimedQuests) {
          const questDef = quests.find((qd) => qd.type === q.quest_type);
          if (questDef) {
            totalPointsEarned += questDef.points;
            db.run(
              `UPDATE user_quests SET claimed = 1 WHERE user_id = ? AND quest_type = ? AND assigned_date = ?`,
              [userId, q.quest_type, today],
              (updateErr) => {
                if (updateErr) console.error(updateErr.message);
              }
            );
          }
        }

        // Check for bonus
        // Ensure all *assigned* quests are completed and unclaimed to get the bonus
        const allAssignedQuests = await new Promise((resolve, reject) => {
          db.all(
            `SELECT * FROM user_quests WHERE user_id = ? AND assigned_date = ?`,
            [userId, today],
            (err, rows) => {
              if (err) return reject(err);
              resolve(rows);
            }
          );
        });

        const allAssignedCompletedAndUnclaimed =
          allAssignedQuests.length === 3 &&
          allAssignedQuests.every((q) => q.completed && !q.claimed);

        if (allAssignedCompletedAndUnclaimed) {
          totalPointsEarned += dailyQuestBonusPoints;
          bonusAwarded = true;
        }

        // Add points to user's total
        db.get(
          `SELECT points FROM users WHERE id = ?`,
          [userId],
          (err, row) => {
            if (err) {
              console.error(err.message);
              return message.channel.send(
                "An error occurred while updating your total points."
              );
            }
            const currentPoints = row ? row.points : 0;
            const newTotalPoints = currentPoints + totalPointsEarned;

            db.run(
              `INSERT OR REPLACE INTO users (id, points) VALUES (?, ?)`,
              [userId, newTotalPoints],
              function (err) {
                if (err) {
                  console.error(err.message);
                  return message.channel.send(
                    "An error occurred while saving your new total points."
                  );
                }
                let response = `🎉 You claimed your quest rewards and earned ${totalPointsEarned} points!`;
                if (bonusAwarded) {
                  response += ` (Including a ${dailyQuestBonusPoints} points bonus for completing all 3!)`;
                }
                response += `\nYour new total points: ${newTotalPoints}.`;
                message.channel.send(response);
              },
              closeDatabase()
            );
          }
        );
      }
    );
  }
// Function to close the database connection.
function closeDatabase() {
    // db.close() closes the database connection.
    // It's important to close the connection when you're done with it to free up resources.
    db.close((err) => {
        if (err) {
            console.error('Error closing database:', err.message);
        } else {
            console.log('Database connection closed.');
        }
    });
}

    dealerDisplayValue = dealerHandValue.value;

    dealerHandDisplay = `${getCardDisplay(

    dealerDisplayValue = calculateHandValue([game.dealerHand[0]]).value;

async function playDealersTurnAndDetermineWinner(
  userId,
  game,
  playerDiscordUser,
  gameMessage,
  initialStatusMessage = ""
) {
  let dealerHandValue = calculateHandValue(game.dealerHand);
  let statusMessage = initialStatusMessage;
  let dealerActions = []; // To store dealer's hits descriptions

  // 1. Reveal dealer's hidden card (this is one intentional update)
  statusMessage += `Dealer reveals their hand.\n`;
  let revealEmbed = getBlackjackEmbed(
    game,
    playerDiscordUser,
    true,
    statusMessage,
    false
  );
  await gameMessage.edit({
    embeds: [revealEmbed],
    components: revealEmbed.components,
  });
  await new Promise((resolve) => setTimeout(resolve, 1000)); // Short pause after reveal

  // 2. Dealer hits until 17 or more (Collect all actions without updating embed)
  while (dealerHandValue.value < 17) {
    const newCard = drawCard(game.deck);
    game.dealerHand.push(newCard); // Add card to hand
    dealerHandValue = calculateHandValue(game.dealerHand); // Recalculate hand value

    // Only store the action string, DO NOT update embed here.
    dealerActions.push(
      `Dealer hits and draws ${getCardDisplay(newCard)}. Dealer's hand is now ${
        dealerHandValue.value
      }.`
    ); // Keep detail for final display
  }

  // 3. Add all collected dealer actions to the status message (ONLY ONCE)
  if (dealerActions.length > 0) {
    statusMessage += "\n" + dealerActions.join("\n"); // Join all hit actions
    statusMessage += `\nDealer's final hand is ${dealerHandValue.value}.\n`; // Concise final hand summary
  } else {
    statusMessage += "Dealer stands.\n"; // If dealer didn't hit at all
  }

  // 4. Determine final winner and outcome text
  let finalWinner = null;
  const finalPlayerHandValue = calculateHandValue(game.playerHand); // Recalculate player hand value for comparison

  if (dealerHandValue.value > 21) {
    statusMessage += "Dealer busted! You win!";
    finalWinner = "player";
  } else if (finalPlayerHandValue.value > dealerHandValue.value) {
    statusMessage += "You win!";
    finalWinner = "player";
  } else if (dealerHandValue.value > finalPlayerHandValue.value) {
    statusMessage += "Dealer wins!";
    finalWinner = "dealer";
  } else {
    statusMessage += "It's a push!";
    finalWinner = "push";
  }

  // 5. Final embed update (ONLY ONE UPDATE for all dealer's actions + final result)
  const finalEmbed = getBlackjackEmbed(
    game,
    playerDiscordUser,
    true,
    statusMessage,
    true
  ); // Game is now over
  await gameMessage.edit({
    embeds: [finalEmbed],
    components: finalEmbed.components,
  });

  // Handle payout and clear game state
  await handleBlackjackPayout(userId, finalWinner, game.betAmount, gameMessage);
  delete activeBlackjackGames[userId]; // Clear game state
}
async function revealRouletteResult(channelId) {
  const game = activeRouletteGames[channelId];
  if (!game) return;

  const channel = client.channels.cache.get(channelId);
  if (!channel) {
    console.error(
      `Could not find channel ${channelId} to reveal roulette result.`
    );
    delete activeRouletteGames[channelId];
    return;
  }
  const allBets = game.bets;
  const spinResult = getRouletteSpinResult(); // Call the new function
  const winningNumber = spinResult.winningNumber;
  const winningColor = spinResult.winningColor;
  const winningSlot = spinResult.winningSlot;

  let winnersList = [];
  let losersList = [];
  let totalWon = 0;
  let totalLost = 0;

  for (const bet of allBets) {
    let userWonThisBet = false;
    let winMultiplier = 0;

    const normalizedBetType = bet.betType; // Use the normalized bet type from the stored bet
    const numBet = parseInt(normalizedBetType);
    if (
      normalizedBetType === numBet.toString() &&
      !isNaN(numBet) &&
      numBet >= 0 &&
      numBet <= 36
    ) {
      if (winningNumber === numBet) {
        userWonThisBet = true;
        winMultiplier = payoutMultipliers.number;
      }
    } else if (["red", "black"].includes(normalizedBetType)) {
      if (winningNumber !== 0 && winningColor === normalizedBetType) {
        userWonThisBet = true;
        winMultiplier = payoutMultipliers[normalizedBetType];
      }
    } else if (["odd", "even"].includes(normalizedBetType)) {
      if (
        winningNumber !== 0 &&
        winningSlot.isOdd === (normalizedBetType === "odd") &&
        winningSlot.isEven === (normalizedBetType === "even")
      ) {
        userWonThisBet = true;
        winMultiplier = payoutMultipliers[normalizedBetType];
      }
    } else if (["low", "high"].includes(normalizedBetType)) {
      if (
        winningNumber !== 0 &&
        winningSlot.isLow === (normalizedBetType === "low") &&
        winningSlot.isHigh === (normalizedBetType === "high")
      ) {
        userWonThisBet = true;
        winMultiplier = payoutMultipliers[normalizedBetType];
      }
    } else if (["1st", "2nd", "3rd"].includes(normalizedBetType)) {
      if (winningSlot.dozen === normalizedBetType) {
        userWonThisBet = true;
        winMultiplier = payoutMultipliers[normalizedBetType];
      }
    } else if (["col1", "col2", "col3"].includes(normalizedBetType)) {
      if (winningSlot.column === normalizedBetType) {
        userWonThisBet = true;
        winMultiplier = payoutMultipliers[normalizedBetType];
      }
    }

    if (userWonThisBet) {
      const winnings = bet.betAmount * winMultiplier;
      await updateUserBalance(bet.userId, winnings);
      winnersList.push(
        `<@${bet.userId}> won ${nexus_coin} **${winnings}** (bet ${bet.betAmount} on ${normalizedBetType})`
      );
      totalWon += winnings;
    } else {
      // User already had betAmount deducted at time of placing bet.
      // No need to deduct again.
      losersList.push(
        `<@${bet.userId}> lost ${nexus_coin} **${bet.betAmount}** (bet ${bet.betAmount} on ${bet.betType})`
      );
      totalLost += bet.betAmount;
    }
  }
  const resultEmbed = new EmbedBuilder()
    .setColor(
      winningColor === "red"
        ? 0xff0000
        : winningColor === "black"
        ? 0x000000
        : 0x00ff00
    ) // Color based on winning number
    .setTitle("<:rouletteWheel:1397521672243777566> Roulette Round Results!")
    .setDescription(
      `The ball landed on **${winningNumber}** (${winningColor})!`
    )
    .addFields(
      {
        name: "Winners 🎉",
        value:
          winnersList.length > 0
            ? winnersList.join("\n")
            : "No winners this round.",

async function handleBlackjackPayout(userId, winner, betAmount, gameMessage) {
  let payoutAmount = 0;
  let payoutMessage = "";

  if (winner === "player") {
    const game = activeBlackjackGames[userId]; // Get game to check for Blackjack payout
    const playerHandValue = calculateHandValue(game.playerHand);

    if (playerHandValue.isBlackjack) {
      payoutAmount = betAmount * 2.5; // 3:2 payout (1.5 profit + 1 original = 2.5 total)
      payoutMessage = `You won ${nexus_coin} **${payoutAmount}** with Blackjack!`;
    } else {
      payoutAmount = betAmount * 2; // 1:1 payout (1 profit + 1 original = 2 total)
      payoutMessage = `You won ${nexus_coin} **${payoutAmount}** !`;
    }
    await updateUserBalance(userId, payoutAmount); // Add total payout
  } else if (winner === "push") {
    payoutAmount = betAmount; // Return original bet
    payoutMessage = `It's a push! Your bet of ${nexus_coin} **${betAmount}** has been returned.`;
    await updateUserBalance(userId, payoutAmount); // Return bet
  } else {
    // Dealer wins (player loses)
    payoutMessage = `You lost your bet of ${nexus_coin} **${betAmount}**.`;
    // Bet was already deducted when starting the game, so no further action needed here for loss
  }

  const newBalance = await getUserBalance(userId); // Get updated balance
  // payoutMessage += ` Your new balance is ${nexus_coin} **${newBalance}**.`;

  // Update the embed with the payout message
  const currentEmbed = gameMessage.embeds[0];
  const finalEmbed = EmbedBuilder.from(currentEmbed)
    .setColor(
      winner === "player" ? 0x00ff00 : winner === "dealer" ? 0xff0000 : 0xffff00
    ) // Set final color
    .setFields(
      // Keep existing fields
      ...currentEmbed.fields,
      // Add a new field for the payout result and new balance
      { name: "Final Result", value: payoutMessage, inline: false },
      {
        name: "Your New Balance",
        value: `${nexus_coin} ${newBalance}`,
        inline: false,
      }
    )
    .setFooter({ text: `Blackjack! Game Over.` }) // Clear original footer or make it final
    .setTimestamp();
  await gameMessage.edit({
    embeds: [finalEmbed],
    components: [],
  });
}

async function handleBlackjackPayout(userId, winner, betAmount, gameMessage) {
  const nexus_coin = "<:nexus_bronze_coin:1396438179288191007>";
  const game = activeBlackjackGames[userId] || {}; // Use dummy if game cleared prematurely
  const playerHandValue = calculateHandValue(game.playerHand || []); // Ensure playerHandValue is defined for checks below

  let netChange = 0;
  let outcomeText = "";

  if (winner === "player") {
    if (playerHandValue.isBlackjack) {
      netChange = betAmount * 2.5; // Profit part of 3:2 payout
      outcomeText = "Blackjack!";
    } else {
      netChange = betAmount * 2; // Profit part of 1:1 payout
      outcomeText = "Win!";
    }
    await updateUserBalance(userId, netChange + betAmount); // Add profit PLUS original bet back
  } else if (winner === "push") {
    netChange = 0;
    outcomeText = "Push!";
    await updateUserBalance(userId, betAmount); // Return original bet
  } else {
    // Dealer wins (player loses)
    netChange = -betAmount;
    outcomeText = "Loss!"; // General loss. Specific statusMessage from hit/stand will describe "Bust!"
  }

  const newBalance = await getUserBalance(userId);

  // Build the final result line (e.g., "Result: Bust 💰 -500")
  let resultLine = `Result: **${outcomeText}** ${nexus_coin} **${
    netChange > 0 ? "+" : ""
  }${netChange}**`;

  // Get the current embed from the message to retain previous information
  const currentEmbed = gameMessage.embeds[0];

  const finalGameEmbed = EmbedBuilder.from(currentEmbed)
    .setColor(
      winner === "player" ? 0x00ff00 : winner === "dealer" ? 0xff0000 : 0xffff00
    ) // Set final color
    // Combine status message from dealer's turn AND the new result line in description
    .setDescription(`${currentEmbed.description}\n\n${resultLine}`) // Added double newline for spacing
    .setFields(currentEmbed.fields) // Carry over existing fields
    .setFooter({
      text: `Blackjack! Game Over.`,
    })
    .setTimestamp();

  await gameMessage.edit({
    embeds: [finalGameEmbed],
    components: [], // Clear components as game is over
  });
}

      if (game.playerHand.length !== 2) {
        return await interaction.followUp({
          content: "You can only Double Down on your first two cards!",
          ephemeral: true,
        });
      }
      if (playerBalance < doubleBetAmount) {
        // Check if player has enough to double
        return await interaction.followUp({
          content: `You don't have enough to double down! You need ${nexus_coin} ${doubleBetAmount} more.`,
          ephemeral: true,
        });
      }

      // 2. Deduct the additional bet
      await updateUserBalance(userId, -doubleBetAmount);
      game.betAmount += doubleBetAmount; // Update game's total bet amount for payout calculation

      // 3. Draw exactly one card
      const newCard = drawCard(game.deck);
      game.playerHand.push(newCard);
      const playerHandValue = calculateHandValue(game.playerHand);

      let statusMessage = `You doubled down and drew ${getCardDisplay(
        newCard
      )}. Your hand is now ${playerHandValue.value}.\n`;
      let gameOver = false;
      let finalWinner = null;

      if (playerHandValue.value > 21) {
        statusMessage += "You busted! Your hand is over 21. You lose!";
        finalWinner = "dealer";
        gameOver = true;
      }

      // 4. Update the embed to show current state after the draw
      const updatedEmbedDoubleDown = getBlackjackEmbed(
        game,
        interaction.user, // Use interaction.user as the playerDiscordUser
        false, // Dealer card still hidden for this display
        statusMessage,
        gameOver
      );
      await game.gameMessage.edit({
        embeds: [updatedEmbedDoubleDown],
        components: updatedEmbedDoubleDown.components,
      });

      // 5. Automatically Stand (proceed to dealer's turn if not busted)
      if (gameOver) {
        // If busted immediately after double down
        await handleBlackjackPayout(
          userId,
          finalWinner,
          game.betAmount,
          game.gameMessage
        );
        delete activeBlackjackGames[userId];
      } else {
        // If not busted, proceed to dealer's turn
        await playDealersTurnAndDetermineWinner(
          userId,
          game,
          interaction.user,
          game.gameMessage,
          statusMessage
        );

const sqlite3 = require('sqlite3').verbose();
const path = require('path');

// Connect to the database
// The database file will be created in the same directory as your bot's main script
const dbPath = path.resolve(__dirname, 'aqua.db');
const db = new sqlite3.Database(dbPath, (err) => {
    if (err) {
        console.error('Error connecting to database:', err.message);
    } else {
        console.log('Connected to the SQLite database.');
    }
});

// Initialize the database: Create the 'users' table if it doesn't exist
function initDb() {
    return new Promise((resolve, reject) => {
        db.run(`
            CREATE TABLE IF NOT EXISTS users (
                user_id TEXT PRIMARY KEY,
                balance INTEGER DEFAULT 1000
            )
        `, (err) => {
            if (err) {
                console.error('Error creating users table:', err.message);
                reject(err);
            } else {
                console.log('Users table checked/created successfully.');
                resolve();
            }
        });
    });
}

// Get user balance
function getUserBalance(userId) {
    return new Promise((resolve, reject) => {
        // Insert user if they don't exist, then retrieve balance
        db.run('INSERT OR IGNORE INTO users (user_id) VALUES (?)', [userId], function(err) {
            if (err) {
                console.error('Error inserting new user:', err.message);
                return reject(err);
            }
            db.get('SELECT balance FROM users WHERE user_id = ?', [userId], (err, row) => {
                if (err) {
                    console.error('Error getting user balance:', err.message);
                    reject(err);
                } else {
                    resolve(row ? row.balance : 1000); // Should always find a row due to INSERT OR IGNORE
                }
            });
        });
    });
}

// Update user balance
function updateUserBalance(userId, amount) {
    return new Promise((resolve, reject) => {
        db.run('UPDATE users SET balance = balance + ? WHERE user_id = ?', [amount, userId], function(err) {
            if (err) {
                console.error('Error updating user balance:', err.message);
                reject(err);
            } else if (this.changes === 0) {
                // This means the user_id wasn't found, which shouldn't happen if getUserBalance is called first
                // but good to handle defensively.
                reject(new Error(`User with ID ${userId} not found for balance update.`));
            } else {
                resolve();
            }
        });
    });
}
function deleteUser(userId) {
    return new Promise((resolve, reject) => {
        db.run('DELETE FROM users WHERE user_id = ?', [userId], function(err) {
            if (err) {
                console.error('Error deleting user:', err.message);
                reject(err);
            } else {
                // 'this.changes' property indicates the number of rows affected by the last query.
                if (this.changes > 0) {
                    console.log(`User ${userId} deleted successfully.`);
                    resolve(true); // User was found and deleted
                } else {
                    console.log(`User ${userId} not found in database.`);
                    resolve(false); // User was not found
                }
            }
        });
    });
}
